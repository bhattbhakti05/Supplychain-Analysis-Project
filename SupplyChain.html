# Creating function which will split univariate sequence into samples
def create_df(sequence, time_stemp):
    X, Y = [], []
    for i in range(len(sequence)):
        if i+time_stemp > len(sequence)-1:
            break
        seq_x = sequence[i:(i+time_stemp)]
        seq_y=sequence[i + time_stemp]
        X.append(seq_x)
        Y.append(seq_y)
    return np.array(X),  np.array(Y)

model = Sequential()

model.add(Dense(1))
model.compile(optimizer='adam', loss='mse')

model = Sequential() : used to stack multiple layers on top of each other. model.add(LSTM(50, activation='relu', input_shape=(n_steps, n_features))): An LSTM layer is added to the model by this line. ReLU serves as the activation function for the 50 memory cells in the LSTM layer, which also accepts a two-dimensional input with 3 time steps and 1 variable for each step and 50 memory cells. model.add(Dense(1)): This line adds a fully connected layer to the model with a single output. model.compile(optimizer='adam', loss='mse'): This line assembles the model, specifying mean squared error as the training loss function and Adam as the optimizer.
